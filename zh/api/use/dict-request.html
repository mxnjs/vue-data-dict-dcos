# 字典数据请求

#

  • 类型
function(dict: DictMeta): Promise | Array

  • 示例

    • 返回Promise
    {
  request(dictMeta) {
    return new Promise((resolve, reject) => {
      setTimeout(function() {
          resolve([
          { label: 'js', value: 1 },
          { label: 'java', value: 2 },
          { label: 'c', value: 3 },
          ])
      }, 3000)
    })
  },
}
    • 返回Array
    {
  request(dictMeta) {
    return [
      { label: 'js', value: 1 },
      { label: 'java', value: 2 },
      { label: 'c', value: 3 },
    ]
  },
}
